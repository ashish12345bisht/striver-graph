#include <iostream>
#include <vector>
#include <queue>
using namespace std;
int main() {
    int m, n;
    cin>>m>>n;
    vector<int> adj[n+1];
    for(int i=0;i<n;i++){
        int u, v;
        cin>>u>>v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    // for(int i=1;i<n+1;i++){
    //     cout<<i<<" ";
    //     for(int j=0;j<adj[i].size();j++){
    //         cout<<adj[i][j]<<" ";
    //     }
    //     cout<<endl;
    // }
    queue<int> q;
    vector<int> v(n+1,0); 
    q.push(1);
    while(!q.empty()){
        int cur = q.front();
        q.pop();
        if(v[cur]==1)continue;
        v[cur]=1;
        cout<<cur<<" ";
        for(int i=0;i<adj[cur].size();i++){
            int temp = adj[cur][i];
            // v[temp]=1;
            q.push(temp);
            // cout<<temp<<" ";
        }
    }
}
